294 void AirDC::PitotCorrection(int mode)//Based on http://basicairdata.blogspot.it/2014/07/pitot-correction-for-position-and.html
295 {
296  switch (mode)
297  {
298  case 1: //No_compensation
299  {
300  _TASPCorrected=_TAS;
301  break;
302  }
303  case 2: //Steady state(no angular acceleration) assumed for this method
304  {
305  float R[3][3];
306  float PB[3][1]; //Position of probe tip in body coordinates
307  float WB[3][1]; //Angular rates in body coordinates [p q r]'
308  float WW[3][1];//Angular rates in wind coordinates [p q r]'
309  float PW[3][1]; //Position of probe tip in wind ref. frame
310  float PWDOT[3][1]; //Velocity of tip in wind ref. frame
311  float VCorrected[3][1]; //Measured Airspeed
312  PB[0][0]=_PitotXcog; //Installation position respect c.o.g.
313  PB[1][0]=_PitotYcog;
314  PB[2][0]=_PitotZcog;
315  WB[0][0]=_Ip-_AOSdot*sin(_AOA); //Angular rates . P, q, r from sensors and
316  //WB[1][0]=_Iq-_AOAdot;
317  WB[1][0]=0;
318  WB[2][0]=_Ir+_AOSdot*cos(_AOA);
319 //Matrix.Print((float*)PB,3,1,"PB");
320  R[0][0]=cos(_AOA)*cos(_AOS);
321  R[0][1]=sin(_AOS);
322  R[0][2]=sin(_AOA)*cos(_AOS);
323  R[1][0]=-1*cos(_AOA)*sin(_AOS);
324  R[1][1]=cos(_AOS);
325  R[1][2]=-1*sin(_AOA)*sin(_AOS);
326  R[2][0]=-1*sin(_AOA);
327  R[2][1]=0;
328  R[2][2]=cos(_AOA);
329  Matrix.Multiply((float*)R,(float*)PB,3,3,1,(float*)PW);//Calculation of Position vector in wind axes
330  Matrix.Multiply((float*)R,(float*)WB,3,3,1,(float*)WW);//Calculation of angular rates at tip in wind frame.
331  PWDOT[0][0]=WW[1][0]*PW[2][0]-WW[2][0]*PW[1][0];//Calculation of velocity vector at tip in wind coordinates. Cross product WWxPW.
332  PWDOT[1][0]=WW[2][0]*PW[0][0]-WW[0][0]*PW[2][0];
333  PWDOT[2][0]=WW[0][0]*PW[1][0]-WW[1][0]*PW[0][0];
334  VCorrected[0][0]=_TAS-PWDOT[0][0];//Airspeed vector
335  VCorrected[1][0]= -PWDOT[1][0];
336  VCorrected[2][0]= -PWDOT[2][0];
337  _TASPCorrected=sqrt(pow(VCorrected[0][0],2)+pow(VCorrected[1][0],2)+pow(VCorrected[2][0],2));
338  break;
339  }
340  }
